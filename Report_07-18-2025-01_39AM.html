  fat = aov(Fat_g / Calories ~ broad_category, data = df),

qq_plots <- lapply(models, function(m) {
  res <- residuals(m)
  ggplot(data.frame(res), aes(sample = res)) +
    stat_qq() +
    stat_qq_line() +
    ggtitle(paste0("QQ Plot: ", deparse(m$call[[2]]))) +
    theme_minimal()
})

do.call(grid.arrange, c(qq_plots, ncol = 2))
